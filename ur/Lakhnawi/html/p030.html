[٦]ثُمَّ حِكْمَةٌ حَقِّـيِّةٌ فِي كَلِمَةٍ إِسْحَاقِـيَّةٍ.

[٧]ثُمَّ حِكْمَةٌ عَلِيَّةٌ فِي كَلِمَةٍ إِسْمَاعِيْلِـيَّةٍ.

[٨]ثُمَّ حِكْمَةٌ رُوحِـيَّةٌ فِي كَلِمَةٍ يَعْقُوبِـيَّةٍ.

[٩]ثُمَّ حِكْمَةٌ نُورِيَّةٌ فِي كَلِمَةٍ يُوسُفِيَّةٍ.

[٠١] ثُمَّ حِكْمَةٌ أَحَدِيَّةٌ فِي كَلِمَةٍ هُودِيَّةٍ.

[١١] ثُمَّ حِكْمَةٌ فَاتِحِـيَّةٌ فِي كَلِمَةٍ صَالِحِـيَّةٍ.

[٢١] ثُمَّ حِكْمَةٌ قَـلْبِـيَّةٌ فِي كَلِمَةٍ شُعَـيْـبِـيَّةٍ.

[٣١] ثُمَّ حِكْمَةٌ مَلَـكِيَّةٌ فِي كَلِمَةٍ لُوطِيَّةٍ.

[٤١] ثُمَّ حِكْمَةٌ قَدَرِيَّةٌ فِي كَلِمَةٍ عُـزَيْـرِيَّةٍ.

[٥١] ثُمَّ حِكْمَةٌ نَـبَوِيَّةٌ فِي كَلِمَةٍ عِيسَوِيَّةٍ.

[٦١] ثُمَّ حِكْمَةٌ رَحْمَانِـيَّةٌ فِي كَلِمَةٍ سُلَيْمَانِـيَّةٍ.

[٧١] ثُمَّ حِكْمَةٌ وُجُودِيَّةٌ فِي كَلِمَةٍ دَاوُدِيَّةٍ.

[٨١] ثُمَّ حِكْمَةٌ نَفْسِيَّةٌ فِي كَلِمَةٍ يُونُسِيَّةٍ.

[٩١] ثُمَّ حِكْمَةٌ غَـيْـبِـيَّةٌ فِي كَلِمَةٍ أيُّوبِيَّةٍ.

[٠٢] ثُمَّ حِكْمَةٌ جَلَالِيَّةٌ فِي كَلِمَةٍ يَحْـيَوِيَّةٍ.
